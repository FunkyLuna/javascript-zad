4. Przyklady (zapis C#)

Fragmenty przykładowego kodu ilustrujące abstrakcję i dziedziczenie.

// Abstrakcja
abstract class Wielokat....
{
protected int[] boki;


public int IleBokow()
{
return boki.Length;
}


public int Obwod()
{
int obwod = 0;
foreach (int bok in boki)
{
obwod += bok;
}
return obwod;
}


public abstract int Pole();
}


// Dziedziczenie
class Kwadrat : Wielokat
{
public Kwadrat(int a)
{
boki = new int[4];
boki[0] = boki[1] = boki[2] = boki[3] = a;
}


public override int Pole()
{
return boki[0] * boki[0];
}
}

5. Różnice — w skrócie

  • Strukturalne: program zbudowany z funkcji/podprogramów — sekwencyjny przepływ sterowania.
  • Obiektowe: funkcje (metody) powiązane z obiektami — utrudnia przypadkowe przekazanie niewłaściwych zmiennych, lepsza modularność stanu.
